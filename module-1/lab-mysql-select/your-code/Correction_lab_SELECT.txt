/* Challenge 1 : select what titles each author has published at which publishers

SELECT authors.au_id AS 'AUTHOR ID', authors.au_lname AS 'LAST NAME', authors.au_fname AS 'FIRST NAME', titles.title AS 'TITLE', publishers.pub_name AS 'PUBLISHER' 

FROM authors

INNER JOIN titleauthor ON authors.au_id = titleauthor.au_id

LEFT JOIN titles ON titleauthor.title_id= titles.title_id

LEFT JOIN publishers ON titles.pub_id = publishers.pub_id

ORDER BY authors.au_id ;

/* Challenge 2 :  query how many titles each author has published at each publisher: */

SELECT authors.au_id AS 'AUTHOR ID', authors.au_lname AS 'LAST NAME', authors.au_fname AS 'FIRST NAME', titles.title AS 'TITLE', publishers.pub_name AS 'PUBLISHER', COUNT(titles.title_id) AS 'COUNT TITLE'

FROM authors

INNER JOIN titleauthor ON authors.au_id = titleauthor.au_id

LEFT JOIN titles ON titleauthor.title_id= titles.title_id

LEFT JOIN publishers ON titles.pub_id = publishers.pub_id

GROUP BY authors.au_id, publishers.pub_id

ORDER BY authors.au_id DESC;

/* On vérifie que le nombre de livres correspond au nombre d'enregistrements de la table titleauthor */

/* Challenge 3 : Select the top 3 authors who have sold the highest number of titles: */

SELECT authors.au_id AS 'AUTHOR ID', authors.au_lname AS 'LAST NAME', authors.au_fname AS 'FIRST NAME',  SUM(titles.ytd_sales) AS 'TOTAL'
FROM authors

INNER JOIN titleauthor ON authors.au_id = titleauthor.au_id

LEFT JOIN titles ON titleauthor.title_id= titles.title_id

GROUP BY authors.au_lname, authors.au_fname

ORDER BY SUM(titles.ytd_sales) DESC

LIMIT 3;

/* Challenge 4 : On utilise la commande COALESCE pour transformer les valeurs NULL en 0 */

SELECT authors.au_id AS 'AUTHOR ID', authors.au_lname AS 'LAST NAME', authors.au_fname AS 'FIRST NAME', COALESCE(SUM(COALESCE(titles.ytd_sales,0)),0) AS 'TOTAL'
FROM authors

LEFT JOIN titleauthor ON authors.au_id = titleauthor.au_id

LEFT JOIN titles ON titleauthor.title_id= titles.title_id

GROUP BY authors.au_lname, authors.au_fname

ORDER BY SUM(titles.ytd_sales) DESC;
